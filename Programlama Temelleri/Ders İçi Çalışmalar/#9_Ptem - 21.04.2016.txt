            //Eðer baþlangýç deðeri bitiþden büyükse iki deðiþken içerði yer deðiþtirsin.
            int ciftoplam = 0, depo;
            Console.Write("Baþlangýç deðeri gir:");
            int basla = Convert.ToInt16(Console.ReadLine());
            Console.Write("Bitiþ deðeri gir:");
            int bitis = Convert.ToInt16(Console.ReadLine());
            if (basla > bitis) 
            {
                depo = basla;
                basla = bitis;
                bitis = depo;
            }
            do
            {
                if ((basla % 2) == 0) 
                {
                    ciftoplam += basla;
                }
                basla++;
                
            } while (basla <= bitis);
            Console.WriteLine(ciftoplam);
            Console.ReadKey();
